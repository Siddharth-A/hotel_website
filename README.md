# hotel_website
sample hotel website

# FRONT-END
1. npm create vite@latest frontend -- --template react
2. cd frontend
3. npm run dev
4. npm install axios
5. npm install @mui/material @emotion/styled @emotion/react

# BACK-END
0. python3 -m venv venv
1. source venv/bin/activate
2. pip install -r requirements.txt
3. python app.py
